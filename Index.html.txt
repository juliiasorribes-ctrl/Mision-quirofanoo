    <script>
        // --- 1. DATOS DEL JUEGO ---
        const questions = [
            {

            },

            },
            {

            {

                options: ["Preguntándole a su familiar", "Verificando su nombre en la pizarra", "Revisando la pulsera de identificación y preguntándole su nombre completo", "Asumiendo que es el paciente de la cama 4"],

            },
            {
                q: "¿Dónde se deben desechar siempre las hojas de bisturí y agujas?",
                options: ["En la bolsa de basura negra", "En la bolsa de desechos biológicos (roja)", "En el contenedor rígido de cortopunzantes", "Dejarlas en la mesa quirúrgica para que limpieza las bote"],

            },
            {

                options: ["Cubrir solo la boca", "Cubrir boca y nariz y ajustarla al puente nasal", "Llevarla colgada en el cuello hasta que empiece la cirugía", "Usarla solo si estás resfriado"],
                correct: 1
            },
            {
                q: "¿Quién es el encargado principal de realizar el 'Marcaje Quirúrgico' en el sitio a operar?",

                correct: 2
            },
            {
                q: "En cuanto a la circulación y tráfico de personal dentro del quirófano durante una cirugía:",
                options: ["Las puertas deben mantenerse abiertas para ventilar", "Se debe minimizar el tránsito y mantener las puertas cerradas", "Cualquier persona puede entrar si usa mascarilla", "El personal debe entrar y salir rápido sin importar cuántas veces"],
                correct: 1
            },
            {
                q: "El Consentimiento Informado es un documento vital. ¿En qué momento debe estar firmado por el paciente?",
                options: ["Durante la cirugía, cuando esté anestesiado", "Después de la cirugía en recuperación", "Antes de administrar cualquier sedación o anestesia", "No es necesario si la cirugía es de urgencia leve"],

            },
            {
                q: "¿Qué vestimenta es obligatoria para transitar por los pasillos internos (zona limpia) del pabellón?",
                options: ["Ropa de calle", "Traje de mayo (uniforme de pabellón) de uso exclusivo y gorro", "Solo delantal blanco", "Traje de mayo, gorro, y cubrezapatos aunque vengas de la calle"],
                correct: 1
            }
        ];

        // --- 2. VARIABLES DE ESTADO ---
        let currentScore = 100;
        let currentQuestionIndex = 0;
        let playerName = "";

        // Elementos del DOM
        const startScreen = document.getElementById('start-screen');
        const gameScreen = document.getElementById('game-screen');
        const endScreen = document.getElementById('end-screen');
        const scoreDisplay = document.getElementById('score-display');
        const statusEmoji = document.getElementById('status-emoji');
        const questionText = document.getElementById('question-text');
        const optionsContainer = document.getElementById('options-container');
        const currentQuestionNum = document.getElementById('current-question-num');

        // --- 3. FUNCIONES DE LÓGICA ---

        function startGame() {
            const nameInput = document.getElementById('player-name').value.trim();
            if (!nameInput) {
                alert("¡Por favor, ingresa tu nombre para comenzar la misión!");
                return;
            }
            playerName = nameInput;
            currentScore = 100;
            currentQuestionIndex = 0;
            
            updateScoreAndEmoji();
            
            startScreen.classList.add('hidden');
            gameScreen.classList.remove('hidden');
            
            loadQuestion();
        }

        function loadQuestion() {
            const currentQ = questions[currentQuestionIndex];
            questionText.innerText = currentQ.q;
            currentQuestionNum.innerText = currentQuestionIndex + 1;
            
            optionsContainer.innerHTML = ''; // Limpiar opciones anteriores
            
            currentQ.options.forEach((opt, index) => {
                const btn = document.createElement('button');
                btn.className = "bg-blue-50 border-2 border-blue-200 text-blue-900 font-medium py-4 px-6 rounded-xl hover:bg-blue-600 hover:text-white hover:border-blue-600 transition-all duration-200 shadow-sm";
                btn.innerText = opt;
                btn.onclick = () => checkAnswer(index);
                optionsContainer.appendChild(btn);
            });
        }

        function checkAnswer(selectedIndex) {
            const correctIndex = questions[currentQuestionIndex].correct;
            
            if (selectedIndex === correctIndex) {
                currentScore += 10;
            } else {
                currentScore -= 10;
            }
            
            updateScoreAndEmoji();
            
            currentQuestionIndex++;
            if (currentQuestionIndex < questions.length) {
                loadQuestion();
            } else {
                endGame();
            }
        }

        function updateScoreAndEmoji() {
            scoreDisplay.innerText = currentScore;
            
            // Lógica para cambiar emoji
            if (currentScore >= 150) {
                statusEmoji.innerText = "😎";
            } else if (currentScore >= 100) {
                statusEmoji.innerText = "😁";
            } else if (currentScore >= 60) {
                statusEmoji.innerText = "😰";
            } else {
                statusEmoji.innerText = "🚑";
            }
        }

        function endGame() {
            gameScreen.classList.add('hidden');
            endScreen.classList.remove('hidden');
            
            document.getElementById('final-score').innerText = currentScore;
            
            // Mensaje final según puntaje (< 80 regular, >= 80 bien hecho)
            const finalMessage = document.getElementById('final-message');
            if (currentScore >= 80) {
                finalMessage.innerHTML = `¡Bien hecho, <b>${playerName}</b>! Dominas muy bien los protocolos de seguridad.`;
            } else {
                finalMessage.innerHTML = `Regular, <b>${playerName}</b>. Te sugerimos repasar los protocolos para asegurar el bienestar del paciente.`;
            }

            saveAndShowRanking();
        }

        function saveAndShowRanking() {
            // Guardar en localStorage
            let rankings = JSON.parse(localStorage.getItem('quirofanoRanking')) || [];
            rankings.push({ name: playerName, score: currentScore });
            
            // Ordenar de mayor a menor y guardar
            rankings.sort((a, b) => b.score - a.score);
            localStorage.setItem('quirofanoRanking', JSON.stringify(rankings));

            // Mostrar el top 5
            const rankingList = document.getElementById('ranking-list');
            rankingList.innerHTML = '';
            const top5 = rankings.slice(0, 5);
            
            top5.forEach((entry, index) => {
                const li = document.createElement('li');
                li.className = "flex justify-between items-center bg-white p-2 rounded border border-blue-50";
                
                let medal = "🏅";
                if(index === 0) medal = "🥇";
                if(index === 1) medal = "🥈";
                if(index === 2) medal = "🥉";

                li.innerHTML = `<span class="font-medium text-gray-700">${medal} ${entry.name}</span> 
                                <span class="font-bold text-teal-600">${entry.score} pts</span>`;
                rankingList.appendChild(li);
            });
        }

        function resetGame() {
            endScreen.classList.add('hidden');
            startScreen.classList.remove('hidden');
            document.getElementById('player-name').value = '';
        }

        function exitGame() {
            // Reinicia todo visualmente simulando una salida
            resetGame();
            document.getElementById('start-screen').innerHTML = `
                <div class="text-6xl mb-4">👋</div>
                <h1 class="text-3xl font-bold text-teal-700 mb-4">¡Turno terminado!</h1>
                <p class="text-gray-600">Gracias por cuidar de nuestros pacientes hoy.</p>
                <button onclick="location.reload()" class="mt-6 text-teal-600 underline hover:text-teal-800">Volver a iniciar</button>
            `;
        }
    </script>
